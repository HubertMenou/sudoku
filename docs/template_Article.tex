\begin{algorithm}[H]
\caption{\code{SolveSudoku} : Main function}
\begin{algorithmic}
\REQUIRE Sudoku \code{initial\_sudoku}
\STATE \code{queue := SudokuQueue(initial\_sudoku)}
\WHILE{True}
	\STATE \code{current\_sudoku := queue.pop()}
	\STATE \code{(success, solution\_candidate) := DirectSolve(current\_sudoku)}
	\STATE \textbf{if} \code{success} \textbf{then break}
	\STATE \code{queue.MakeAChoiceAndAdd(solution\_candidate)}
\ENDWHILE
\RETURN \code{solution\_candidate}
\end{algorithmic}
\end{algorithm}
